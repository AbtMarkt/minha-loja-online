async function finalizarCompra(){ let carrinho = JSON.parse(localStorage.getItem('carrinho')) || []; if(carrinho.length===0){ alert("Carrinho vazio!"); return; } let nome = document.getElementById('clienteNome').value; let email = document.getElementById('clienteEmail').value; let telefone = document.getElementById('clienteTelefone').value; let pedido = { itens: carrinho, total: carrinho.reduce((sum,i)=>sum+i.preco,0), data: new Date().toLocaleString(), status: "Pendente", cliente: nome, email: email, telefone: telefone }; await fetch('http://localhost:3000/pedidos',{ method:'POST', headers:{ 'Content-Type':'application/json' }, body: JSON.stringify(pedido) }); localStorage.removeItem('carrinho'); mostrarCarrinho(); alert("Compra registrada com sucesso!"); }
